# i18n-Overlay Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Translate the Paperclip GUI into German (and any future language) via swappable JSON dictionaries, delivered as a real Paperclip plugin plus a userscript fallback, without forking the core UI.

**Architecture:** A framework-free DOM translation engine runs inside the host document (slot UIs share the host DOM — verified). It replaces exact-match text nodes and whitelisted attributes against a JSON dictionary and keeps a `MutationObserver` running at page lifetime via a module-level singleton. A Paperclip `sidebar` slot component kicks the singleton off; the same engine is also bundled as a Tampermonkey userscript. A Babel-based dev-only harvest tool seeds the dictionary from `ui/src/**/*.tsx`.

**Tech Stack:** TypeScript, Vitest + jsdom (engine tests), Babel (`@babel/parser`/`@babel/traverse`, harvest), esbuild (userscript bundle), `@paperclipai/plugin-sdk` (plugin glue), `tsc` (build).

**Spec:** `docs/superpowers/specs/2026-06-12-i18n-overlay-design.md`

```ts
export interface Dictionary {
  $meta?: { language: string; version?: number };
  text?: Record<string, string>;
  attr?: Record<string, string>;
}

export interface TranslatorOptions {
  attributes?: string[];
  skipSelectors?: string[];
  root?: HTMLElement;
}

export interface Translator {
  start(): void;
  stop(): void;
  translateTree(root: Node): void;
}

const DEFAULT_ATTRS = ["placeholder", "title", "aria-label"];
const DEFAULT_SKIP = ["script", "style", "code", "pre", "textarea", '[contenteditable="true"]'];

export function createTranslator(dict: Dictionary, options: TranslatorOptions = {}): Translator {
  const text = dict.text ?? {};
  const skip = options.skipSelectors ?? DEFAULT_SKIP;
  // Records the exact value we wrote, so the observer (Task 4) can ignore our own writes.
  const written = new WeakMap<Node, string>();

  function inSkippedSubtree(node: Node): boolean {
    const el = node.nodeType === Node.ELEMENT_NODE ? (node as Element) : node.parentElement;
    if (!el) return false;
    return skip.some((sel) => el.closest(sel) !== null);
  }

  function translateTextNode(node: Text): void {
    const raw = node.nodeValue ?? "";
    const key = raw.trim();
    if (!key) return;
    const value = text[key];
    if (value === undefined || value === key) return;
    if (inSkippedSubtree(node)) return;
    const lead = raw.slice(0, raw.indexOf(key));
    const trail = raw.slice(raw.indexOf(key) + key.length);
    const next = lead + value + trail;
    node.nodeValue = next;
    written.set(node, next);
  }

  function translateTree(root: Node): void {
    const walker = document.createTreeWalker(root, NodeFilter.SHOW_TEXT);
    const nodes: Text[] = [];
    let n = walker.nextNode();
    while (n) { nodes.push(n as Text); n = walker.nextNode(); }
    for (const t of nodes) translateTextNode(t);
  }

  const root = options.root ?? document.body;
  return {
    start() { translateTree(root); },
    stop() { /* observer wired up in Task 4 */ },
    translateTree,
  };
}
```

- [ ] **Step 3: Wire up the observer**

In `engine.ts`, add an `observer` variable in the closure and rewrite `start`/`stop`. The loop guard relies on the `written` WeakMap (already populated in `translateTextNode`): skip a `characterData` mutation when the node's current value equals what we last wrote.

```ts
  let observer: MutationObserver | null = null;

  function handleMutations(records: MutationRecord[]): void {
    for (const rec of records) {
      if (rec.type === "characterData") {
        const node = rec.target as Text;
        if (written.get(node) === node.nodeValue) continue; // our own write
        translateTextNode(node);
      } else if (rec.type === "attributes") {
        if (rec.target.nodeType === Node.ELEMENT_NODE) translateElementAttrs(rec.target as Element);
      } else {
        rec.addedNodes.forEach((added) => {
          if (added.nodeType === Node.TEXT_NODE) translateTextNode(added as Text);
          else if (added.nodeType === Node.ELEMENT_NODE) translateTree(added);
        });
      }
    }
  }

  return {
    start() {
      translateTree(root);
      observer = new MutationObserver(handleMutations);
      observer.observe(root, {
        subtree: true,
        childList: true,
        characterData: true,
        attributes: true,
        attributeFilter: attrs,
      });
    },
    stop() { observer?.disconnect(); observer = null; },
    translateTree,
  };
```

- [ ] **Step 4: Implement the harvest tool**

```ts
import { parse } from "@babel/parser";
import _traverse from "@babel/traverse";
import { readFileSync, writeFileSync } from "node:fs";
import { globSync } from "node:fs";

// @babel/traverse ships as CJS; normalize the default export for ESM.
const traverse = (_traverse as unknown as { default?: typeof _traverse }).default ?? _traverse;

const ATTR_PROPS = new Set(["placeholder", "title", "aria-label", "label"]);

export interface Extracted { text: string[]; attr: string[]; }

export function extractStrings(code: string): Extracted {
  const ast = parse(code, { sourceType: "module", plugins: ["typescript", "jsx"] });
  const text = new Set<string>();
  const attr = new Set<string>();
  traverse(ast, {
    JSXText(path) {
      const v = path.node.value.trim();
      if (v) text.add(v);
    },
    JSXAttribute(path) {
      const name = String(path.node.name.name);
      if (!ATTR_PROPS.has(name)) return;
      const val = path.node.value;
      if (val && val.type === "StringLiteral" && val.value.trim()) attr.add(val.value.trim());
    },
  });
  return { text: [...text], attr: [...attr] };
}

export interface Dict { $meta?: unknown; text: Record<string, string>; attr: Record<string, string>; }

export function mergeInto(existing: Dict, found: Extracted): Dict {
  const out: Dict = { ...existing, text: { ...existing.text }, attr: { ...existing.attr } };
  for (const k of found.text) if (!(k in out.text)) out.text[k] = "";
  for (const k of found.attr) if (!(k in out.attr)) out.attr[k] = "";
  return out;
}

// CLI entry: harvest ui/src and merge into src/dictionary/de.json
function main(): void {
  const uiRoot = new URL("../../../../ui/src/", import.meta.url);
  const files = globSync("**/*.tsx", { cwd: uiRoot });
  const all: Extracted = { text: [], attr: [] };
  for (const rel of files) {
    const code = readFileSync(new URL(rel, uiRoot), "utf8");
    const e = extractStrings(code);
    all.text.push(...e.text);
    all.attr.push(...e.attr);
  }
  const dictUrl = new URL("../src/dictionary/de.json", import.meta.url);
  const existing = JSON.parse(readFileSync(dictUrl, "utf8")) as Dict;
  const merged = mergeInto(existing, all);
  writeFileSync(dictUrl, JSON.stringify(merged, null, 2) + "\n");
  const empties = Object.values(merged.text).filter((v) => v === "").length
    + Object.values(merged.attr).filter((v) => v === "").length;
  console.log(`Harvested ${files.length} files; ${empties} untranslated keys remain.`);
}

if (import.meta.url === `file://${process.argv[1]}`) main();
```

```tsx
import dictionary from "../dictionary/de.json";

/**
 * Invisible sidebar slot. Its only job is to run inside the host document and
 * kick off the page-lifetime translation singleton.
 */
export function I18nOverlayMount(_props: PluginSidebarProps) {
  useEffect(() => {
    ensureStarted(dictionary as Parameters<typeof ensureStarted>[0]);
  }, []);
  return null;
}
```

```
// @run-at       document-idle

```

> Note: adjust `@match` on each machine if Paperclip is bound to a LAN/tailnet host. The localhost matches cover the default `npx paperclipai onboard` bind.

- [ ] **Step 2: Create `src/userscript/entry.ts`**

- [ ] **Step 3: Create `esbuild.userscript.mjs`**

```mjs
import esbuild from "esbuild";
import { readFileSync, writeFileSync } from "node:fs";

const header = readFileSync(new URL("./src/userscript/header.txt", import.meta.url), "utf8");

const result = await esbuild.build({
  entryPoints: ["src/userscript/entry.ts"],
  bundle: true,
  format: "iife",
  platform: "browser",
  target: "es2020",
  loader: { ".json": "json" },
  write: false,
});

const code = result.outputFiles[0].text;
writeFileSync("dist/paperclip-de.user.js", header + "\n" + code);
console.log("Wrote dist/paperclip-de.user.js");
```
