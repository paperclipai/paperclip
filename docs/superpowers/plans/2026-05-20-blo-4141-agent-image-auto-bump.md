# BLO-4141 Agent Image Auto-Bump Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Eliminate the manual step required to roll out `paperclip-agent` image updates: when `docker-agent.yml` builds a new image, automatically bump every eligible agent's `adapter_config.image`, skipping in-flight agents and auto-retrying them on next run completion.

**Architecture:** New `POST /api/admin/agents/bump-agent-image` admin endpoint owns the policy: enumerate eligible agents (`adapter_type IN ('claude_k8s','opencode_k8s')`), check in-flight via heartbeat_runs + k8s BatchV1Api, idempotent-PATCH the idle ones (delegating to the existing `agents.ts:2626` audit-logged flow), and stash a `pending_image_bump` value for the busy ones. A hook in `heartbeat.ts setRunStatus` retries the bump when a run reaches a terminal state. The workflow is dumb — one `curl` after the image build.

**Tech Stack:** TypeScript + Express 5 (server), Drizzle ORM + Postgres (db), Zod (validation), Vitest (tests), `@kubernetes/client-node` (in-cluster API), GitHub Actions (`workflow_run` trigger).

```typescript
import { agents } from "@paperclipai/db/schema/agents";
import { and, eq, inArray, isNotNull, sql } from "drizzle-orm";

const ELIGIBLE_ADAPTER_TYPES = ["claude_k8s", "opencode_k8s"] as const;

export interface EligibleAgent {
  id: string;
  companyId: string;
  name: string;
  adapterType: string;
  currentImage: string;
}

export async function selectEligibleAgentsForImageBump(args: {
  companyId: string;
  targetImage: string;
}): Promise<EligibleAgent[]> {
  const rows = await db
    .select({
      id: agents.id,
      companyId: agents.companyId,
      name: agents.name,
      adapterType: agents.adapterType,
      adapterConfig: agents.adapterConfig,
    })
    .from(agents)
    .where(
      and(
        eq(agents.companyId, args.companyId),
        inArray(agents.adapterType, [...ELIGIBLE_ADAPTER_TYPES]),
        isNotNull(sql`${agents.adapterConfig}->>'image'`),
        sql`${agents.adapterConfig}->>'image' <> ${args.targetImage}`,
      ),
    );

  return rows.map((row) => ({
    id: row.id,
    companyId: row.companyId,
    name: row.name,
    adapterType: row.adapterType,
    currentImage: (row.adapterConfig as Record<string, unknown>).image as string,
  }));
}
```

```typescript
import { heartbeatRuns } from "@paperclipai/db/schema/heartbeat-runs";
import { hasActiveJobForAgent } from "./k8s-job-liveness.js";

const IN_FLIGHT_RUN_STATUSES = ["queued", "running"] as const;

export async function isAgentInFlight(agentId: string): Promise<boolean> {
  // DB check first (cheap; covers the well-behaved case).
  const [dbHit] = await db
    .select({ id: heartbeatRuns.id })
    .from(heartbeatRuns)
    .where(
      and(
        eq(heartbeatRuns.agentId, agentId),
        inArray(heartbeatRuns.status, [...IN_FLIGHT_RUN_STATUSES]),
      ),
    )
    .limit(1);

  if (dbHit) return true;

  // k8s confirm catches ghost Jobs whose heartbeat_runs row was already
  // marked terminal (e.g. timeout reaper) but the pod is still running.
  return hasActiveJobForAgent(agentId);
}
```

```typescript
const AGENT_ID_LABEL = "paperclip.io/agentId";

/**
 * Returns true if the cluster has any non-terminal Job for this agent.
 *
 * Used by the agent-image-bump service as a second-pass confirmation
 * after the heartbeat_runs DB check — catches the case where a Job
 * outlives its DB row (timeout reaper marked the run failed but the
 * pod is still running). Falls back to `false` if the k8s client is
 * unavailable (local dev, or cluster API unreachable); the DB check
 * is authoritative for the common case.
 */
export async function hasActiveJobForAgent(agentId: string): Promise<boolean> {
  const state = initClient();
  if (state.kind !== "ready") return false;

  try {
    const res = await state.api.listNamespacedJob({
      namespace: PAPERCLIP_K8S_NAMESPACE,
      labelSelector: `${AGENT_JOB_LABEL_SELECTOR},${AGENT_ID_LABEL}=${agentId}`,
    });
    const items = res.items ?? [];
    return items.some((job) => {
      const status = job.status;
      if (!status) return true; // Treat unknown status as in-flight to be safe.
      const active = status.active ?? 0;
      const succeeded = status.succeeded ?? 0;
      const failed = status.failed ?? 0;
      // active > 0 => still has running pods. If succeeded === 0 && failed === 0,
      // the Job is freshly created and hasn't been scheduled yet — also in-flight.
      return active > 0 || (succeeded === 0 && failed === 0);
    });
  } catch (error) {
    const reason = error instanceof Error ? error.message : String(error);
    logger.warn({ agentId, error: reason }, "k8s in-flight check failed; falling back to DB-only");
    return false;
  }
}
```

```typescript
import { logger } from "../middleware/logger.js";

export interface ApplyResult {
  agentId: string;
  outcome: "bumped" | "skipped";
}

export async function applyImageBumpToAgent(args: {
  agentId: string;
  targetImage: string;
  /** Free-form source tag for audit trail. e.g. "ci:docker-agent.yml" or "auto-retry-on-completion". */
  source: string;
}): Promise<ApplyResult> {
  const inFlight = await isAgentInFlight(args.agentId);
  if (inFlight) {
    await db
      .update(agents)
      .set({ pendingImageBump: args.targetImage, updatedAt: new Date() })
      .where(eq(agents.id, args.agentId));
    logger.info(
      { agentId: args.agentId, targetImage: args.targetImage, source: args.source },
      "agent in-flight; pending_image_bump set",
    );
    return { outcome: "skipped", agentId: args.agentId };
  }

  await patchAgentImage(args.agentId, args.targetImage, args.source);
  return { outcome: "bumped", agentId: args.agentId };
}

async function patchAgentImage(agentId: string, targetImage: string, source: string): Promise<void> {
  // Read current adapterConfig, splice image, write back. Mirrors the
  // partial-patch semantics of PATCH /agents/:id (server/src/routes/agents.ts:2626)
  // without going through HTTP.
  const [row] = await db.select({ adapterConfig: agents.adapterConfig }).from(agents).where(eq(agents.id, agentId));
  if (!row) throw new Error(`agent ${agentId} not found`);
  const existing = (row.adapterConfig as Record<string, unknown>) ?? {};
  const next = { ...existing, image: targetImage };

  await db.transaction(async (tx) => {
    await tx
      .update(agents)
      .set({ adapterConfig: next, pendingImageBump: null, updatedAt: new Date() })
      .where(eq(agents.id, agentId));
    // Audit entry mirrors the shape that the PATCH route writes (see
    // agents.ts revision-log helper). The "source" column distinguishes
    // CI-driven bumps from interactive PATCHes.
    await tx.execute(sql`
      INSERT INTO agent_config_revisions
        (id, company_id, agent_id, source, changed_keys, before_config, after_config, created_at)
      SELECT
        gen_random_uuid(), ${agents.companyId}, ${agentId}, ${source},
        ARRAY['adapterConfig']::text[],
        jsonb_build_object('adapter_config', ${existing}::jsonb),
        jsonb_build_object('adapter_config', ${next}::jsonb),
        now()
      FROM ${agents} WHERE ${agents.id} = ${agentId}
    `);
  });

  logger.info({ agentId, targetImage, source }, "agent image PATCHed");
}
```

```typescript
export interface BumpBatchSummary {
  bumped: string[];
  skipped: string[];
  unchanged: string[];
}

export async function bumpAgentImagesForCompany(args: {
  companyId: string;
  targetImage: string;
  source: string;
}): Promise<BumpBatchSummary> {
  // Identify the already-on-target agents BEFORE the eligibility filter
  // strips them, so the response can call them out explicitly.
  const unchanged = await db
    .select({ id: agents.id })
    .from(agents)
    .where(
      and(
        eq(agents.companyId, args.companyId),
        inArray(agents.adapterType, [...ELIGIBLE_ADAPTER_TYPES]),
        sql`${agents.adapterConfig}->>'image' = ${args.targetImage}`,
      ),
    );

  const candidates = await selectEligibleAgentsForImageBump({
    companyId: args.companyId,
    targetImage: args.targetImage,
  });

  const bumped: string[] = [];
  const skipped: string[] = [];

  for (const candidate of candidates) {
    const result = await applyImageBumpToAgent({
      agentId: candidate.id,
      targetImage: args.targetImage,
      source: args.source,
    });
    if (result.outcome === "bumped") bumped.push(result.agentId);
    else skipped.push(result.agentId);
  }

  return { bumped, skipped, unchanged: unchanged.map((r) => r.id) };
}
```

```typescript
export async function processPendingImageBumpForAgent(agentId: string): Promise<void> {
  const [row] = await db
    .select({ pending: agents.pendingImageBump, adapterConfig: agents.adapterConfig })
    .from(agents)
    .where(eq(agents.id, agentId));

  if (!row || !row.pending) return;

  // If the agent's current image already matches the pending target, just clear
  // the column. This can happen if the same image was applied via another path
  // (e.g. operator PATCH) between the original skip and the retry.
  const currentImage = (row.adapterConfig as Record<string, unknown>).image;
  if (currentImage === row.pending) {
    await db.update(agents).set({ pendingImageBump: null }).where(eq(agents.id, agentId));
    return;
  }

  // Honor in-flight: if another run is already queued / running, defer.
  if (await isAgentInFlight(agentId)) {
    logger.info({ agentId }, "pending image bump deferred; another run is in-flight");
    return;
  }

  await patchAgentImage(agentId, row.pending, "auto-retry-on-completion");
}
```

```typescript
import { type Router } from "express";
import { z } from "zod";
import { validate } from "../../middleware/validate.js";
import { assertCompanyAccess } from "../../auth/company-access.js";
import { forbidden } from "../../errors.js";
import { bumpAgentImagesForCompany } from "../../services/agent-image-bump.js";

const bumpAgentImageSchema = z.object({
  companyId: z.string().uuid(),
  image: z
    .string()
    .min(1)
    .max(500)
    .regex(/^[\w./@:-]+$/, "image must be a valid container reference"),
  source: z.string().min(1).max(120).optional().default("admin:manual"),
  buildSha: z.string().min(1).max(40).optional(),
});

export function registerBumpAgentImageRoute(router: Router): void {
  router.post(
    "/admin/agents/bump-agent-image",
    validate(bumpAgentImageSchema),
    async (req, res) => {
      const body = req.body as z.infer<typeof bumpAgentImageSchema>;

      // Only board tokens with agents:create permission may bump.
      // Agent-actor tokens are deliberately excluded — bumping is a
      // platform-ops action, not an agent action.
      if (req.actor.type !== "board") {
        throw forbidden("Board token required for bump-agent-image");
      }
      assertCompanyAccess(req, body.companyId);

      const source = body.buildSha ? `${body.source}@${body.buildSha}` : body.source;
      const summary = await bumpAgentImagesForCompany({
        companyId: body.companyId,
        targetImage: body.image,
        source,
      });

      res.json(summary);
    },
  );
}
```
