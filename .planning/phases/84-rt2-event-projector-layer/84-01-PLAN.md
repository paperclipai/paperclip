export function rt2DomainEventService(db: Db) {
  // append: idempotency-safe insert — if idempotencyKey exists and record found, returns existing
  async function append(input: AppendRt2DomainEvent): Promise<Rt2DomainEvent>

  // appendAndProject: append + projector processEvent for rt2.activity_live_bridge + rt2KnowledgeProjectorService
  async function appendAndProject(input: AppendRt2DomainEvent): Promise<Rt2DomainEvent>

  // processEvent: projector state machine — skips if already processed, updates projector state
  async function processEvent(
    projectorName: string,
    eventId: string,
    handler: (event: Rt2DomainEvent) => Promise<void>,
  ): Promise<{ status: "processed" | "skipped" }>

  // projectActivityAndLive: writes to activity_log + publishes live event
  async function projectActivityAndLive(event: Rt2DomainEvent): Promise<void>
}

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Runtime → Database | Runtime service writes state changes atomically with domain event emission |
| API → appendAndProject | Untrusted input crosses at AppendRt2DomainEvent schema validation (zod) |
| Projector → Read Model | Projectors update read models — failure must not corrupt projector state cursor |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-84-01 | Tampering | rt2-task-execution.ts | mitigate | Idempotency key unique index on (companyId, idempotencyKey) prevents duplicate events; append() checks existing before insert (line 96-110 of rt2-domain-events.ts) |
| T-84-02 | Denial | processEvent projector loop | accept | Projector skips already-processed events via rt2V33ProjectorEvents lookup — retry-safe, no duplicate processing |
| T-84-03 | Information Disclosure | rt2-task-engine.ts | mitigate | RT2-native event contracts use typed Rt2DomainEventType union; appendRt2DomainEventSchema validates eventType against allowed list |
| T-84-04 | Repudiation | Projector state (failed) | mitigate | failureCount increments on each failure; threshold mechanism exists for manual intervention trigger |
| T-84-05 | Elevation | Execution state transitions | mitigate | State check constraint in DB: `rt2_v33_execution_attempts_state_check` restricts state values; executorType check restricts to ('user', 'jarvis', 'runtime') |
</threat_model>
