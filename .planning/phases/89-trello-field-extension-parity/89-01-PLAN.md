    ```typescript
      await assertIssue(companyId, issueId);
      await db.insert(rt2V33TaskParticipants).values({
        companyId, taskIssueId: issueId, userId, state: 'active',
        joinedByUserId: actorUserId, joinedAt: new Date()
      }).onConflictDoUpdate({
        target: [rt2V33TaskParticipants.taskIssueId, rt2V33TaskParticipants.userId],
        set: { state: 'active', endedAt: null, endedReason: null }
      }).returning();
    },

    removeMemberFromCard: async (companyId: string, issueId: string, participantId: string) => {
      await db.update(rt2V33TaskParticipants).set({
        state: 'ended', endedAt: new Date(), endedReason: 'self_left'
      }).where(and(
        eq(rt2V33TaskParticipants.companyId, companyId),
        eq(rt2V33TaskParticipants.id, participantId)
      ));
    },
    ```
