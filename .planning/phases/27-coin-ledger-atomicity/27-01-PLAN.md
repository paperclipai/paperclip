  ELSE 'debit'

**Replace `recordCoinTransaction` in `server/src/services/rt2-personal-pnl.ts`** (lines 341-379):

```typescript
async function recordCoinTransaction(
  companyId: string,
  fromActorId: string,
  fromActorType: string,
  toActorId: string,
  toActorType: string,
  amount: number,
  transactionType: CoinLedgerEntry["transactionType"],
  description: string,
  referenceId?: string,
  referenceType?: string,
  period?: string,
): Promise<CoinLedgerEntry> {
  const p = period || getCurrentPeriod();

  // Determine leg: positive amount = credit (增加), negative = debit (减少)
  const leg: "credit" | "debit" = amount >= 0 ? "credit" : "debit";

  // Compute balanceAfter atomically via SQL subquery — no read-then-write race
  const [entry] = await db
    .insert(rt2CoinLedger)
    .values({
      companyId,
      fromActorId,
      fromActorType,
      toActorId,
      toActorType,
      amount,
      balanceAfter: sql<number>`(
        SELECT COALESCE(SUM(${rt2CoinLedger.amount}), 0) + ${amount}
        FROM ${rt2CoinLedger}
        WHERE ${rt2CoinLedger.companyId} = ${companyId}
          AND ${rt2CoinLedger.toActorId} = ${toActorId}
          AND ${rt2CoinLedger.toActorType} = ${toActorType}
      )`,
      transactionType,
      description,
      referenceId: referenceId ?? null,
      referenceType: referenceType ?? null,
      period: p,
      leg,
    })
    .returning();

  return entry as CoinLedgerEntry;
}
```

**Key changes:**
1. `balanceAfter` now uses a SQL subquery in the INSERT VALUES — atomic, no race
2. `leg` is computed in application code and inserted with each entry
3. `getActorBalance()` is still used for READS only — never in write path
4. The subquery `COALESCE(SUM(...), 0) + amount` computes what the balance WILL be after this entry
