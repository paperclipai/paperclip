async function recordIncome(
  companyId: string,
  actorId: string,
  actorType: "user" | "agent",
  amount: number,
  description: string,
  referenceId?: string,
  referenceType?: string,
  period?: string,
): Promise<PersonalPnL> {
  const p = period || getCurrentPeriod();

  // Atomic: P&L update + ledger entry in single transaction
  const result = await db.transaction(async (tx) => {
    // Get or create P&L within transaction
    const pnlRows = await tx
      .select()
      .from(rt2PersonalPnL)
      .where(
        and(
          eq(rt2PersonalPnL.companyId, companyId),
          eq(rt2PersonalPnL.actorId, actorId),
          eq(rt2PersonalPnL.actorType, actorType),
          eq(rt2PersonalPnL.period, p),
        ),
      )
      .then((rows) => rows[0] ?? null);

    let pnl: PersonalPnL;
    if (pnlRows) {
      pnl = pnlRows as PersonalPnL;
    } else {
      const [created] = await tx
        .insert(rt2PersonalPnL)
        .values({
          companyId,
          actorId,
          actorType,
          period: p,
          income: 0,
          expenses: 0,
          netPnL: 0,
          budgetAllocated: 0,
          budgetUsed: 0,
        })
        .returning();
      pnl = created as PersonalPnL;
    }

    // Update P&L
    const newIncome = pnl.income + amount;
    const newNetPnL = newIncome - pnl.expenses;
    await tx
      .update(rt2PersonalPnL)
      .set({
        income: newIncome,
        netPnL: newNetPnL,
        updatedAt: new Date(),
      })
      .where(eq(rt2PersonalPnL.id, pnl.id));

    // Record in coin ledger (now atomic via Plan 01's subquery)
    const [entry] = await tx
      .insert(rt2CoinLedger)
      .values({
        companyId,
        fromActorId: actorId,
        fromActorType: actorType,
        toActorId: actorId,
        toActorType: actorType,
        amount,
        balanceAfter: sql<number>`(
          SELECT COALESCE(SUM(${rt2CoinLedger.amount}), 0) + ${amount}
          FROM ${rt2CoinLedger}
          WHERE ${rt2CoinLedger.companyId} = ${companyId}
            AND ${rt2CoinLedger.toActorId} = ${actorId}
            AND ${rt2CoinLedger.toActorType} = ${actorType}
        )`,
        transactionType: "earned",
        description,
        referenceId: referenceId ?? null,
        referenceType: referenceType ?? null,
        period: p,
        leg: amount >= 0 ? "credit" : "debit",
      })
      .returning();

    return { ...pnl, income: newIncome, netPnL: newNetPnL };
  });

  return result;
}

async function transferCoins(
  companyId: string,
  fromActorId: string,
  fromActorType: "user" | "agent",
  toActorId: string,
  toActorType: "user" | "agent",
  amount: number,
  description: string,
): Promise<{ from: PersonalPnL; to: PersonalPnL }> {
  // Atomic: deduct from sender + add to receiver in single transaction
  const result = await db.transaction(async (tx) => {
    // Step 1: Get sender P&L + deduct
    const fromRows = await tx
      .select()
      .from(rt2PersonalPnL)
      .where(
        and(
          eq(rt2PersonalPnL.companyId, companyId),
          eq(rt2PersonalPnL.actorId, fromActorId),
          eq(rt2PersonalPnL.actorType, fromActorType),
          eq(rt2PersonalPnL.period, getCurrentPeriod()),
        ),
      )
      .then((rows) => rows[0] ?? null);

    if (!fromRows) throw new Error("Sender P&L not found");
    const fromPnL = fromRows as PersonalPnL;
    const newFromExpenses = fromPnL.expenses + amount;
    const newFromNetPnL = fromPnL.income - newFromExpenses;
    await tx
      .update(rt2PersonalPnL)
      .set({ expenses: newFromExpenses, netPnL: newFromNetPnL, budgetUsed: newFromExpenses, updatedAt: new Date() })
      .where(eq(rt2PersonalPnL.id, fromPnL.id));

    // Step 2: Get or create receiver P&L + add
    const toRows = await tx
      .select()
      .from(rt2PersonalPnL)
      .where(
        and(
          eq(rt2PersonalPnL.companyId, companyId),
          eq(rt2PersonalPnL.actorId, toActorId),
          eq(rt2PersonalPnL.actorType, toActorType),
          eq(rt2PersonalPnL.period, getCurrentPeriod()),
        ),
      )
      .then((rows) => rows[0] ?? null);

    let toPnL: PersonalPnL;
    if (toRows) {
      toPnL = toRows as PersonalPnL;

      const [created] = await tx
        .insert(rt2PersonalPnL)
        .values({
          companyId,
          actorId: toActorId,

          income: 0,
          expenses: 0,
          netPnL: 0,
          budgetAllocated: 0,
          budgetUsed: 0,
        })
        .returning();
      toPnL = created as PersonalPnL;
    }

    const newToIncome = toPnL.income + amount;
    const newToNetPnL = newToIncome - toPnL.expenses;
    await tx
      .update(rt2PersonalPnL)
      .set({ income: newToIncome, netPnL: newToNetPnL, updatedAt: new Date() })
      .where(eq(rt2PersonalPnL.id, toPnL.id));

      .insert(rt2CoinLedger)

        fromActorType,
        toActorId,

/**
 * LEDGER-03: Cross-table P&L reconciliation
 * Compares rt2CoinLedger sum vs rt2PersonalPnL net for an actor/period.
 * Returns diff (should be 0 if tables are in sync).
 */
async function reconcileActorPnL(
  companyId: string,
  actorId: string,
  actorType: "user" | "agent",
  period?: string,
): Promise<{ pnlNet: number; ledgerSum: number; diff: number; isReconciled: boolean }> {
  const p = period || getCurrentPeriod();

  // Get P&L net from rt2PersonalPnL
  const pnlRows = await db
    .select({ income: rt2PersonalPnL.income, expenses: rt2PersonalPnL.expenses })
    .from(rt2PersonalPnL)
    .where(
      and(
        eq(rt2PersonalPnL.companyId, companyId),
        eq(rt2PersonalPnL.actorId, actorId),
        eq(rt2PersonalPnL.actorType, actorType),
        eq(rt2PersonalPnL.period, p),
      ),
    )
    .then((rows) => rows[0] ?? null);

  const pnlNet = pnlRows ? (pnlRows.income ?? 0) - (pnlRows.expenses ?? 0) : 0;

  // Get sum from rt2CoinLedger (credits minus debits = net)
  const ledgerSum = await db
    .select({ total: sql<number>`COALESCE(SUM(${rt2CoinLedger.amount}), 0)` })
    .from(rt2CoinLedger)
    .where(
      and(
        eq(rt2CoinLedger.companyId, companyId),
        eq(rt2CoinLedger.toActorId, actorId),
        eq(rt2CoinLedger.toActorType, actorType),
        eq(rt2CoinLedger.period, p),
      ),
    )
    .then((rows) => Number(rows[0]?.total) || 0);

  const diff = pnlNet - ledgerSum;
  return { pnlNet, ledgerSum, diff, isReconciled: diff === 0 };
}
